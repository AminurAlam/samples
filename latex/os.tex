\begin{center}
  {\Huge \textbf{Common Linux Commands}}
\end{center}
\section{File Management}
\gloss{touch}{Creates a new empty file or updates the access time of an existing file}{touch <filename>}
\gloss{rm}{Removes files, used with -i to prompt before removing something}{rm [-f] [-i] <filename>}
\gloss{cp}{Copies files from one place to another, the destination can be another file or a directory, if destination is a file but multiple sources are given the files get overwritten}{cp <source> <destination> }
\gloss{mv}{Moves files from one place to another, when the destination is a file it acts like a rename}{mv [-i] <source> <destination>}
\gloss{cat}{Concatenates files and displays them, commonly used with a single file to view its contents}{cat <filename>}
\gloss{chmod}{Changes the access permissions for a file. Use + to allow and - to deny a permission followed by the type of permission you want to change}{chmod <+|-><r|w|x> <filename>}
\section{Directory Management}
\gloss{mkdir}{Creates a new empty directory, used with -p to create nested directories}{mkdir [-p] <directory>}
\gloss{rmdir}{Removes an empty directory, the command fails if it has any content inside it, used with -p to remove the parent directory if its empty too}{rmdir [-p] <directory>}
\gloss{rm -r}{Removes directory and all of its contents recursively, used with -f to skip all prompts or -I to only prompt once}{rm -r [-f] [-I] <directory>}
\gloss{ls}{Lists files in a directory, used with -a to also show hidden files and -l to show longer and detailed view. When no directory is provides it lists the files in current working directory}{ls [-l] [-a] <directory>}
\gloss{cd}{Changes the current working directory. If no parameter is given it puts you in \$HOME, or if - is given it goes to previous directory}{cd <directory>}
\gloss{pwd}{Prints the current working directory}{pwd}
\section{Printing to Terminal}
\gloss{echo}{Prints a message to the terminal. Can also be used to print \$variables}{echo <string>}
\gloss{grep}{Searches for a regex pattern in file and prints lines with matches}{grep [-E] /patnern/ <filename>}
\gloss{sed}{Used to perform text transformations like replacement, deletion or addition using regex patterns}{sed s/pattern/replacement/g <filename>}
\gloss{awk}{A programming language commonly used to print specified columns of a file}{awk [-F <seperator>] "\{print \$1 \$3\}" <filename>}
\gloss{find}{Searches for files in a given directory's hierarchy based on multiple filters given to it. Optionally it can execute a command on each matching file}{find <directory> [-name /pattern/] [-type <f|d>] [-exec <command>]}
\gloss{head/tail}{Prints first/last 10 lines of a file or number of lines specified by -n}{<head|tail> [-n <count>] <filename>}
\gloss{seq}{Prints sequence of numbers to the terminal based on given range}{seq [first [[increment] last]]}
\section{System Management}
\gloss{ps}{Displays information about currently running processes}{ps [a][u][x]}
\gloss{kill/pkill}{Stops a process identified by its pid/name by sending a kill signal}{<kill|pkill> <pid|process name>}
\gloss{ping}{Sends data packets to networks, commonly used to test network connection strength}{ping [-c <count>] [-i <interval>] [-s <packetsize>] <host>}
\gloss{date}{Display date and time in a given format}{date [--iso-8601] [+format]}
\gloss{free}{Display amount of free and used memory in the system, used with -h to make the output more readable}{free [-h]}
\gloss{ifconfig}{Displays networks that your device is connected to}{ifconfig [-a] [-s]}
\section{Other}
\gloss{man}{Displays help manual page of a command on how to use it}{man <command>}
\gloss{df}{Shows total/used/available disk space}{df [-h] [-a]}
\gloss{diff}{Compares two files line by line and outputs their differences}{diff [-u] [-r]}
\gloss{tar}{Archiving/compression tool to save multiple files into single tape archive}{tar <x|c>zf <filename>}
\gloss{ln}{Used to create files linked to each other, commonly used to create a shortcut of a file in a different location}{ln [-s] <filename> <destination>}
\gloss{less}{Display the contents of a file in a browsable window}{less <filename>}
\gloss{vi}{A multi modal text editor built with keyboard centric editing in mind}{vi <filename>}
